 1-1 משנה א
מֹשֶׁה קִבֵּל תּוֹרָה מִסִּינַי,
וּמְסָרָהּ לִיהוֹשֻׁעַ,
וִיהוֹשֻׁעַ לִזְקֵנִים,
וּזְקֵנִים לִנְבִיאִים,
וּנְבִיאִים מְסָרוּהָ לְאַנְשֵׁי כְנֶסֶת הַגְּדוֹלָה.
הֵם אָמְרוּ שְׁלֹשָׁה דְבָרִים:
הֱיוּ‏ מְתוּנִין בַּדִּין,
וְהַעֲמִידוּ‏ תַלְמִידִים‏ הַרְבֵּה,
וַעֲשׁוּ סִיָג לַתּוֹרָה. 

 1-2 משנה ב‏ 
שִׁמְעוֹן הַצַּדִּיק הָיָה מִשִּׁירֵי‏ כְנֶסֶת הַגְּדוֹלָה.
הוּא הָיָה אוֹמֵר:
עַל שְׁלֹשָׁה דְבָרִים הָעוֹלָם עוֹמֵד:
עַל הַתּוֹרָה,
וְעַל הָעֲבוֹדָה,
וְעַל גְּמִילוּת חֲסָדִים. 

 1-3 משנה ג
אַנְטִיגְנַס‏ אִישׁ סוֹכוֹ קִבֵּל מִשִּׁמְעוֹן הַצַּדִּיק.
הוּא הָיָה אוֹמֵר:
אַל תִּהְיוּ כַעֲבָדִים הַמְשַׁמְּשִׁים אֶת הָרַב
עַל מְנָת לְקַבֵּל פְּרָס 
אֶלָּא הֱיוּ‏ כַעֲבָדִים הַמְשַׁמְּשִׁים אֶת הָרַב
עַל מְנָת שֶׁלֹּא לְקַבֵּל פְּרָס‏ .
וִיהֵא‏ מוֹרָא שָׁמַיִם עֲלֵיכֶם. 

 1-4 משנה ד
יוֹסֵה‏ בֶן יוֹעֶזֶר אִישׁ צְרֵידָה‏ ,
וְיוֹסֵה בֶן יוֹחָנָן אִישׁ יְרוּשָׁלַיִם קִבְּלוּ מִמֶּנּוּ‏ .
יוֹסֵה בֶן יוֹעֶזֶר אִישׁ צְרֵדָה‏ אוֹמֵר:
יְהִי בֵיתָךְ בֵּית וַעַד לַחֲכָמִים,
וֶהֱוֵי מִתְאַבֵּק בַּעֲפַר רַגְלֵיהֶם,
וְשׁוֹתֶה‏ בְצִמְאָה‏ אֶת דִּבְרֵיהֶם. 

1-5 משנה ה
יוֹסֵה‏ בֶן יוֹחָנָן אִישׁ יְרוּשָׁלַיִם אוֹמֵר:
יְהִי בֵיתָךְ פָּתוּחַ לָרְוָחָה,
וְיִהְיוּ עֲנִיִּים בְּנֵי בֵיתֶךָ,
וְאַל תַּרְבֶּה שִׂיחָה עִם הָאִשָּׁה. 
בְּאִשְׁתּוֹ‏ אָמָרוּ,
קַל וָחֹמֶר בְּאֵשֶׁת חֲבֵרוֹ.
מִכָּן אָמְרוּ חֲכָמִים:
כָּל הַמַּרְבֶּה‏ שִׂיחָה עִם הָאִשָּׁה‏ סופר נוסף: כְשֶׁהִיא נִדָּה‏ ,
גּוֹרֵם רָעָה עַל עַצְמוֹ‏ וּבָטֵל‏ מִדִּבְרֵי תוֹרָה,
וְסוֹפוֹ יוֹרֵשׁ לְגֵיהִנָּם‏. 


 1-6 משנה ו
יְהוֹשֻׁעַ בֶּן פְּרַחְיָה וּמַתַּי‏ הָאַרְבְּלִי קִבְּלוּ מֵהֶן‏ .
יְהוֹשֻׁעַ בֶּן פְּרַחְיָה אוֹמֵר:
עֲשֵׂה לָךְ רַב,
וּקְנֵה לָךְ חָבֵר,
וֶהֱוִי דָן אֶת כָּל הָאָדָן לְכַף זְכוּת. 

 1-7 משנה ז
מַתַּיי‏ הָאַרְבְּלִי אוֹמֵר:
רְחַק סופר: הַרְחֵק‏ מִשָּׁכֵן רַע,
וְאַל תִּתְחַבַּר‏ לָרָשָׁע,
וְאַל‏ תִּתְיוֹאַשׁ‏ מִן‏ הַפֻּרְעָנוּת. 


 1-8 משנה ח
יְהוּדָה בֶן טָבַי וְשִׁמְעוֹן בֶּן שָׁטָח‏ קִבְּלוּ מֵהֶן.
יְהוּדָה בֶן טָבַי אוֹמֵר:
אַל תַּעַשׂ עַצְמָךְ כְּעַרְכֵי‏ סופר: אַרְכִי‏ דַּיָּנִים‏  
וּכְשֶׁיִּהְיוּ‏ בַעֲלֵי הַדִּין עוֹמְדִין לְפָנֶיךָ,
יִהְיוּ‏ בְעֵינֶיךָ כָרְשָׁעִים 
וּכְשֶׁנִּפְטָרִים מִלְּפָנֶיךָ,
יְהוּא בְעֵינֶיךָ כַצַּדִּיקִים,
שֶׁקִּבְּלוּ עֲלֵיהֶן אֶת הַדִּין. 


 1-9 משנה ט
שִׁמְעוֹן‏ בֶּן שָׁטָח אוֹמֵר:
הֱוִי מַרְבֶּה לַחְקוֹר אֶת הָעֵדִים 
וֶהְוִי זָהִיר בִּדְבָרֶיךָ,
שֶׁמֵּא‏ מִתּוֹכָן יִלְמְדוּ לְשַׁקֵּר. 

 1-10 משנה י
שְׁמַעְיָה וְאַבְטַלְיוֹן קִבְּלוּ מֵהֶן.
שְׁמַעְיָה אוֹמֵר:
אֱהֹב אֶת הַמְּלָאכָה,
וּשְׂנָא אֶת הָרַבָּנוּת,
וְאַל תִּתְוַדַּע לָרְשׁוּת‏ . 


 1-11 משנה יא
אַבְטַלְיוֹן‏ אוֹמֵר:
חֲכָמִים, הִזָּהֲרוּ בְדִבְרֵיכֶם,
שֶׁמֵּא תָחוֹבוּ חוֹבַת גָּלוּת,
וְתִגְלוּ לִמְקוֹם הַמַּיִם‏ הָרָעִים,
וְיִשְׁתּוּ הַתַּלְמִידִים הַבָּאִים אַחֲרֵיכֶם וְיָמוּתוּ,
וְנִמְצָא שֵׁם שָׁמַיִם מִתְחַלֵּל. 

1-12 משנה יב
הֶלֵּל וְשַׁמַּי קִבְּלוּ מֵהֶן.
הֶלֵּל אוֹמֵר:
הֱוִי תַלְמִידוֹ‏ שֶׁלְּאַהֲרֹן,
אוֹהֵב שָׁלוֹם וְרוֹדֵף שָׁלוֹם,
אוֹהֵב אֶת הַבְּרִיּוֹת וּמְקָרְבָן לַתּוֹרָה. 

 1-13 משנה יג‏ 
הוּא הָיָה אוֹמֵר:
נְגַד שְׁמָא אָבַד שְׁמָא‏ ,
וְדִילָא‏ מוֹסִיף יֵיסוּף סופר: יסיף‏ ,
וְדִילָא‏ יְלִיף קְטָלָה‏ חַיָּב‏ ,
 סופר נוסף: לָא יַעֲבֵד קְטָלָה קָטְלִין ,
וְדִי‏ יִשְׁתַּמַּשׁ בְּתָגָא חָלַף. 

 1-14 משנה יד‏ 
הוּא הָיָה אוֹמֵר:
אֵין אֲנִי‏ לי , מִי לִי?
וּכְשֶׁאֲנִי לְעַצְמִי, מָה אָנִי?
אִם‏ לֹא עַכְשָׁיו, אֶמָּתַי? 

 1-15 משנה טו‏ 
שַׁמַּי‏ אוֹמֵר:
עֲשֵׂה תוֹרָתָךְ קְבַע,
אֱמֹר מְעַט וַעֲשֵׂה הַרְבֵּה,
וֶהֱוִי מְקַבֵּל אֶת כָּל הָאָדָם בְּסֵבֶר פָּנִים יָפוֹת. 

 1-16 משנה טז 
רַבָּן גַּמְלִיאֵל אוֹמֵר:
עֲשֵׂה לָךְ רַב,
וְהִסְתַּלֵּק‏ מִן הַסָּפֵק,
וְאַל תַּרְבֶּה לְעַשֵּׂר עֲמָדוֹת סופר: אוֹמָדוֹת/עוֹמָדוֹת‏ 

1-17 משנה יז‏ 
שִׁמְעוֹן‏ בְּנוֹ אוֹמֵר:
כָּל יָמַי גָּדַלְתִּי בֵין הַחֲכָמִים,
וְלֹא מָצָאתִי לַגּוּף טוֹב אֶלָּא שְׁתִיקָה.
וְלֹא הַמִּדְרָשׁ הוּא הָעִקָּר,
אֶלָּא הַמַּעֲשֶׂה.
וְכָל הַמַּרְבֶּה דְבָרִים מֵבִיא חֵטְא. 

 1-18 משנה יח‏ 
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר:
עַל שְׁלֹשָׁה דְבָרִים הָעוֹלָם קַיָּם:
עַל הַדִּין וְעַל הָאֱמֶת וְעַל הַשָּׁלוֹם‏ .
 שֶׁנֶּאֱמַר: (זכריה ח, טז)
  אֱמֶת וּמִשְׁפַּט שָׁלוֹם שִׁפְטוּ בְשַׁעֲרֵיכֶם   .

 2-1 משנה א
רְבִּי אוֹמֵר:
אֵיזוֹהִי דֶרֶךְ יְשָׁרָה שֶׁיָּבֹר לוֹ הָאָדָם?
כָּל שֶׁהִיא תִפְאֶרֶת לְעוֹשָׂהּ‏ וְתִפְאֶרֶת‏ לוֹ מִן הָאָדָם.
וֶהֱוֵי זָהִיר בְּמִצְוָה קַלָּה כְמִצְוָה חֲמוּרָה‏ ,
שֶׁאֵין אַתָּה יוֹדֵעַ מַתַּן שְׂכָרָן שֶׁלַּמִּצְוֹת‏ .
וֶהֱוֵי מְחַשֵּׁב הֶפְסֵד מִצְוָה כְנֶגֶד שְׂכָרָהּ,
וּשְׂכַר עֲבֵרָה כְנֶגֶד הֶפְסֵדָהּ.
הִשְׂתַּכֵּל‏ בִּשְׁלֹשָׁה דְבָרִים,
וְאֵין אַתָּה בָא לִידֵי עֲבֵרָה 
דַּע מַה לְּמַעְלָה מִמָּךְ:
עַיִן רוֹאָה,
וְאֹזֶן שׁוֹמַעַת,
וְכָל מַעֲשֶׂיךָ בְּסֵפֶר נִכְתָּבִים.

2-2 משנה ב
רַבָּן גַּמְלִיאֵל בְּנוֹ שֶׁלִרְבִּי יְהוּדָה הַנָּשִׂיא אוֹמֵר:
יָפֶה תַלְמוּד תּוֹרָה עִם דֶּרֶךְ אֶרֶץ,
שֶׁיְּגִיעַת שְׁנֵיהֶם מְשַׁכְּחַת עָוֹן.
וְכָל תּוֹרָה שֶׁאֵין עִמָּהּ מְלָאכָה,
סוֹף‏ בְּטֵילָה לְגָרֵר‏ עָוֹן.
וְכָל הָעֲמֵלִים עִם הַצִּבּוּר,
יְהוּא עֲמֵלִים עִמָּהֶן לְשֵׁם שָׁמַיִם,
שֶׁזְּכוּת אֲבוֹתָם מְסַיַּעַתָּם וְצִדְקָתָם עוֹמֶדֶת לְעַד‏ .
וְאַתֶּם‏ , מַעֲלִים עֲלֵיהֶם נ.א. עֲלֵיכֶם‏ שָׂכָר‏ ,
כְּאִלּוּ עֲשִׂיתֶם.

2-3 משנה ג
הֱווֹן זְהִירִים‏ . בָּרְשׁוּת,
שֶׁאֵין מְקָרְבִין‏ לְאָדָם אֶלָּא לְצֹרֶךְ עַצְמָן 
נִרְאִין בְּאוֹהֲבִין‏ בְּשָׁעַת הֲנָיָתָן,
וְאֵין עוֹמְדִין‏ לְאָדָם בְּשָׁעַת דֹּחֲקוֹ.

2-4 משנה ד‏ 
הוּא הָיָה אוֹמֵר‏ :
עֲשֵׂה רְצוֹנוֹ כִרְצוֹנָךְ‏ ,
‏ שֶׁיַּעֲשֶׂה רְצוֹנָךְ כִּרְצוֹנוֹ.
בַּטֵּל רְצוֹנָךְ מִפְּנֵי רְצוֹנוֹ‏ ,
‏ שֶׁיְּבַטֵּל רְצוֹן אֲחֵרִים מִפְּנֵי רְצוֹנָךְ.

2-5 משנה ה
רְבִּי‏ הֶלֵּל אוֹמֵר:
אַל תִּפְרֹשׁ מִן הַצִּבּוּר 
וְאַל תַּאֲמִין‏ בְּעַצְמָךְ עַד יוֹם מוֹתָךְ 
וְאַל תָּדִין אֶת חֲבֵרָךְ עַד שֶׁתַּגִּיעַ לִמְקוֹמוֹ 
וְאַל תֹּאמַר דָּבָר שֶׁאֵי אֶפְשָׁר לוֹ לְהִשָּׁמַע‏ ,
שֶׁסּוֹפוֹ לְהִשָּׁמֵעַ 
וְאַל תֹּאמַר:
  כְּשֶׁאֶפָּנֶה‏ אֶשְׁנֶה  ,
שֶׁמֵּא לֹא תִפָּנֶה.
הוּא הָיָה אוֹמֵר:
אֵין בּוּר יָרֵא חֵטְ,
וְלֹא עַם הָאָרֶץ חָסִיד,
וְלֹא הַבּוֹיְשָׁן‏ לָמֵד,
וְלֹא הַקּוֹפְדָן‏ מְלַמֵּד,
וְלֹא כָל הַמַּרְבֶּה בִסְחוֹרָה‏ מַחְכִּים,
וּבִמְקוֹם שֶׁאֵין‏ אֲנָשִׁים, הִשְׁתַּדֵּל‏ לִהְיוֹת אִישׁ.

2-6 משנה ו

אַף הוּא רָאָה גֻלְגֹּלֶת אַחַת‏ צָפָה‏ עַל פְּנֵי הַמַּיִם.
וְאָמַר‏ לָהּ:
עַל דְּאַטִיפְתְּ, אֲטִיפוּךְ,
וְסוֹף‏ מַטִּיפַיִךְ יְטוּפוּן.

2-7 משנה ז‏ 
הוּא הָיָה אוֹמֵר:
מַרְבֶּה בָשָׂר מַרְבֶּה רִמָּה 
מַרְבֶּה נְכָסִים מַרְבֶּה דָיוֹן‏

רְבִּי יוֹחָנָן בֶּן בְּרוֹקָה אוֹמֵר:
כָּל הַמְחַלֵּל שֵׁם שָׁמַיִם בַּסֵּתֶר,
נִפְרָעִים מִמֶּנּוּ בַגָּלוּי.
אֶחָד שׁוֹגֵג וְאֶחָד‏ מֵזִיד בְּחִלּוּל הַשֵּׁם.

4-5 משנה ה
רְבִּי יִשְׁמָעֵאל‏ אוֹמֵר:
הַלָּמֵד עַל מְנָת לְלַמֵּד‏ ,
מַסְפִּיקִין בְּיָדוֹ לִלְמֹד וּלְלַמֵּד‏  
הַלָּמֵד עַל מְנָת לַעֲשׂוֹת,
מַסְפִּיקִין בְּיָדוֹ לִלְמֹד וּלְלַמֵּד,
לִשְׁמוֹר‏ וְלַעֲשׁוֹת.
רְבִּי צָדוֹק אוֹמֵר:
אַל תַּעֲשֵׂם עֲטָרָה לְהִתְגַּדֵּל בָּהֶן,
וְלֹא קוֹרְדּוֹם‏ לֹאכַל מֵהֶן‏ .
וְכָךְ‏ הָיָה הֶלֵּל אוֹמֵר:
וְדִי יִשְׁתַּמַּשׁ‏ בְּתָגָא חָלַף.
הָא כָּל‏ הַנֵּאוֹת מִדִּבְרֵי תוֹרָה,
נָטַל חַיָּיו מִן הָעוֹלָם.

4-6 משנה ו‏ 
רְבִּי יוֹסֵה אוֹמֵר:
כָּל הַמְכַבֵּד אֶת הַתּוֹרָה,
גּוּפוֹ מְכֻבָּד מִן נ.א. עַל‏ הַבְּרִיּוֹת 
וְכָל הַמְחַלֵּל אֶת הַתּוֹרָה,
גּוּפוֹ מְחֻלָּל מִן נ.א. עַל‏ הַבְּרִיּוֹת.

4-7 משנה ז‏ 
רְבִּי יִשְׁמָעֵאל בְּנוֹ‏ אוֹמֵר:
הַחוֹסֵךְ‏ עַצְמוֹ מִן הַדִּין,
פּוֹרֵק מִמֶּנּוּ אֵיבָה וְגָזֵל וּשְׁבוּעַת שָׁוְא.
וְהַגַּס לִבּוֹ בְהוֹרָיָה‏ ,
שׁוֹטֶה, וְרָשָׁע, וְגַס רוּחַ.


4-8 משנה ח
הוּא הָיָה אוֹמֵר:
אַל‏ תְּהִי דָן יְחִידִי‏ ,
שֶׁאֵין דָּן יָחִיד‏ אֶלָּא אֶחָד.
וְאַל תֹּאמַר:   קַבְּלוּ‏ דַעְתִּי!  
שֶׁהֵן‏ רַשָּׁיִין וְלֹא אַתָּה.

4-9 משנה ט
רְבִּי יוֹנָתָן אוֹמֵר:
כָּל הַמְקַיֵּם אֶת הַתּוֹרָה מֵעֹנִי,
סוֹפוֹ לְקַיְּמָהּ מֵעֹשֶׁר 
וְכָל הַמְבַטֵּל אֶת הַתּוֹרָה מֵעֹשֶׁר,
סוֹפוֹ לְבַטְּלָהּ מֵעֹנִי.

4-10 משנה י
רְבִּי מֵאִיר אוֹמֵר:
הֱוֵי מְעַט‏ עֵסֶק, וַעֲסֹק בַּתּוֹרָה 
וּשְׁפַל‏ רוּחַ בִּפְנֵי כָל הָאָדָם 
וְאִם בָּטַלְתָּה מִן הַתּוֹרָה,
יֶשׁ לוֹ‏ בְטֵלִים הַרְבֵּה כְנֶגְדָּךְ 
וְאִם עָמַלְתָּ בַתּוֹרָה,
יֶשׁ לוֹ‏ שָׂכָר הַרְבֵּה לִתֶּן לָךְ‏ .

4-11 משנה יא
רְבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹמֵר:
הָעוֹשֶׂה מִצְוָה אַחַת,
קָנָה לוֹ פָרַקְלִיט אֶחָד 
וְהָעוֹבֵר עֲבֵרָה אַחַת‏ ,
קָנָה לוֹ קָטֵיגוֹר אֶחָד.
תְּשׁוּבָה וּמַעֲשִׂים טוֹבִים,
כִּתְרִיס לִפְנֵי‏ הַפֻּרְעָנוּת.
רְבִּי יוֹחָנָן הַסַּנְדַּלָּר אוֹמֵר:
כָּל כְּנֵסְיָה‏ שֶׁהִיא לְשֵׁם‏ שָׁמַיִם‏ ,
סוֹפָהּ לְהִתְקַיֵּם 
וְשֶׁאֵינָה לְשֵׁם שָׁמַיִם‏ ,
אֵין סוֹפָהּ לְהִתְקַיֵּם.

4-12 משנה יב
רְבִּי אֶלְעָזָר‏ אוֹמֵר:
יְהִי כְבוֹד תַּלְמִידָךְ חָבִיב עָלֶיךָ כִכְבוֹד חֲבֵרָךְ‏ ,
וּכְבוֹד חֲבֵרָךְ כְּמוֹרָא רַבָּךְ,
וּמוֹרָא רַבָּךְ כְּמוֹרָא שָׁמַיִם.

4-13 משנה יג
רְבִּי יְהוּדָה אוֹמֵר:
הֱוֵי‏ זָהִיר בַּתַּלְמוּד,
שֶׁשִּׁגְגַת‏ הַתַּלְמוּד עוֹלָה זָדוֹן.
רְבִּי שִׁמְעוֹן אוֹמֵר:
שְׁלֹשָׁה כְתָרִים הֵן:
כֶּתֶר תּוֹרָה,
וְכֶתֶר‏ כְּהֻנָּה,
וְכֶתֶר מַלְכוּת 
וְכֶתֶר שֵׁם טוֹב עַל גַּבֵּיהֶן‏ .

4-14 משנה יד‏ 
רְבִּי נְהוֹרַי אוֹמֵר:
הֱוֵי גוֹלֶה לִמְקוֹם תּוֹרָה,
וְאַל תֹּאמַר   הִיא‏ תָּבוֹא אַחֲרַי  ,
שֶׁחֲבֵרֶיךָ‏ יְקַיְּמוּהָ‏ בְיָדֶךָ.
וְאֶל בִּינָתְךָ אַל תִּשָּׁעֵן.

4-15 משנה טו
רְבִּי יַנַּי אוֹמֵר:
אֵין בְּיָדֵינוּ לֹא מִשַּׁלְוַת הָרְשָׁעִים
וְלֹא נ.א.   וְאַף לֹא‏ מִיִּסּוּרֵיהֶן שֶׁלַּצַּדִּיקִים‏ .
מַתְּיָה‏ בֶן חָרָשׁ אוֹמֵר:
הֱוֵי מַקְדִים לִשְׁלוֹם כָּל‏ הָאָדָם,
וֶהֱוֵי זָנָב לָאֲרָיוֹת,
וְלֹא רֹאשׁ לַשּׁוּעָלִים.

4-16 משנה טז
רְבִּי עֲקִיבָה נ.א. יַעֲקֹב‏ אוֹמֵר:
הָעוֹלָם הַזֶּה דוֹמֶה לִפְרוֹזְדוֹד‏ לִפְנֵי‏ הָעוֹלָם הַבָּא,
הַתְקֵן עַצְמָךְ לִפְרוֹזְדוֹד‏ ,
כְּדֵי שֶׁתִּכָּנֵס לִיטְרִיקְלִין‏ .

4-17 משנה יז
הוּא הָיָה אוֹמֵר:
יָפָה שָׁעָה אַחַת בִּתְשׁוּבָה וּבְמַעֲשִׂים טוֹבִים בָּעוֹלָם הַזֶּה,
מִכָּל חַיֵּי‏ סופר: כְּחַיֵי‏ הָעוֹלָם הַבָּא.
יָפָה‏ שָׁעָה אַחַת שֶׁלְּקוֹרַת רוּחַ בָּעוֹלָם הַבָּא,
מִכָּל חַיֵּי הָעוֹלָם הַזֶּה.

4-18 משנה יח
רְבִּי שִׁמְעוֹן‏ אוֹמֵר:
אַל תְּרַצֶּה אֶת חֲבֵרָךְ בְּשָׁעַת כַּעְסוֹ,
וְאַל תְּנַחֲמֶנּוּ בְּשָׁעָה שֶׁמֵּתוֹ מוּטָל לְפָנָיו,
וְאַל תִּשְׁאַל לוֹ בְּשָׁעַת נִדְרוֹ,
וְאַל תִּשְׁתַּדֵּל‏ סופר: תִּשְׁתַּדַּר‏ לִרְאוֹתוֹ‏ בְּשָׁעַת קַלְקָלָתוֹ.

4-19 משנה יט
שְׁמוּאֵל הַקָּטָן אוֹמֵר: (משלי כד, יז)
בִּנְפֹל אוֹיִבְךָ אַל תִּשְׂמָח,
וּבִכָּשְׁלוֹ‏ אַל יָגֵל לִבֶּךָ.  

4-20 משנה כ
אֱלִישָׁע בֶּן אֲבִייָה‏ אוֹמֵר:
הַלָּמֵד יֶלֶד, לְמַה הוּא דּוֹמֶה?
לִדְיוֹ‏ כְתוּבָה עַל נְיָר חָדָשׁ.
הַלָּמֵד זָקֵן, לְמַה הוּא דּוֹמֶה?
לִדְיוֹ כְתוּבָה‏ עַל נְיָר מָחוּק.
רְבִּי‏ יוֹסֵה בֶן יְהוּדָה אִישׁ כְּפַר הַבַּבְלִי אוֹמֵר:
הַלָּמֵד‏ מִן הַקְּטַנִּים, לְמַה הוּא דּוֹמֶה?
לְאוֹכֵל עֲנָבִים קֵהוֹת וְשׁוֹתֶה‏ יַיִן מִגִּתּוֹ.
וְהַלָּמֵד‏ מִן הַזְּקֵנִים, לְמַה הוּא דּוֹמֶה?
לְאוֹכֵל‏ עֲנָבִים בְּשֵׁלוֹת‏ וְשׁוֹתֶה‏ יַיִן יָשָׁן.
רְבִּי מֵאִיר‏ אוֹמֵר:
אַל תִּסְתַּכֵּל בַּקִּינְקֵן‏ ,
אֶלָּא בְמָה שֶׁיֶּשׁ בּוֹ‏ .
יֵשׁ קִינְקֵן‏ חָדָשׁ מָלֵא יַיִן‏ יָשָׁן,
וְיָשָׁן שֶׁאֲפִלּוּ חָדָשׁ אֵין בּוֹ.


4-21 משנה כא
רְבִּי‏ אֶלְעָזָר‏ סופר: אֶלִיעֶזֶר‏ הַכַּפָּר‏ אוֹמֵר:
הַקִּנְאָה, וְהַתַּאֲוָה, וְהַכָּבוֹד
מוֹצִיאִין אֶת הָאָדָם מִן הָעוֹלָם.

4-22 משנה כב
הוּא הָיָה אוֹמֵר:
הַיִּלּוֹדִים לָמוּת, וְהַמֵּתִים לִחְיוֹת, וְהַחַיִּים לִידּוֹן,
לֵידַע, וּלְהוֹדִיעַ וּלְהִוָּדַע,
שֶׁהוּא הַיּוֹצֵר‏ , וְהוּא הַבּוֹרֵא,
הוּא הַמֵּבִין, וְהוּא הַדַּיָּן,
הוּא עֵד‏ , הוּא בַּעַל דִּין,
וְהוּא עָתִיד לָדוֹן‏ .
‏ שֶׁאֵין לְפָנָיו לֹא עַוְלָה, וְלֹא שִׁכְחָה,
וְלֹא מַשּׂוֹא פָנִים, וְלֹא מַקַּח שֹׁחַד,
שֶׁהַכֹּל שֶׁלּוֹ.
וְדַע, שֶׁהַכֹּל בָּא‏ בְחֶשְׁבּוֹן.
וְאַל‏ יַבְטִיחָךְ יִצְרָךְ שֶׁהַשְּׁאוֹל‏ בֵּית מָנוֹס‏ ,
שֶׁעַל כָּרְחָךְ‏ אַתָּה נוֹצָר,
וְעַל כָּרְחָךְ אַתָּה נוֹלָד,
וְעַל כָּרְחָךְ אַתָּה‏ חַי,
וְעַל כָּרְחָךְ אַתָּה‏ מֵת,
וְעַל כָּרְחָךְ אַתָּה עָתִיד לִתֵּן דִּין וְחֶשְׁבּוֹן,
לִפְנֵי מֶלֶךְ מַלְכֵי הַמְּלָכִים הַקָּדוֹשׁ בָּרוּךְ הוּא.


5-1 משנה א
בַּעֲשָׂרָה מַאֲמָרוֹת נִבְרָא הָעוֹלָם.
וּמַה תַּלְמוּד‏ ?
וַהֲלֹא בְמַאֲמָר‏ אֶחָד הָיָה‏ יָכוֹל לְהִבָּרוֹת?
אֶלָּא לְהִפָּרַע מִן הָרְשָׁעִים,
שֶׁהֵן מְאַבְּדִין‏ אֶת הָעוֹלָם שֶׁנִּבְרָא בַעֲשָׂרָה מַאֲמָרוֹת,
וְלִתֵּן שָׂכָר טוֹב לַצַּדִּיקִים‏ ,
שֶׁהֵן מְקַיְּמִים‏ אֶת הָעוֹלָם שֶׁנִּבְרָא בַעֲשָׂרָה מַאֲמָרוֹת.

5-2 משנה ב
עֲשָׂרָה דוֹרוֹת מֵאָדָם וְעַד נֹחַ,
לְהוֹדִיעַ כַּמָּה אֶרֶךְ אַפַּיִם לְפָנָיו,
שֶׁכָּל הַדּוֹרוֹת הָיוּ מַכְעִיסִים לְפָנָיו,
עַד שֶׁהֵבִיא עֲלֵיהֶם אֶת‏ הַמַּבּוּל.
עֲשָׂרָה דוֹרוֹת מִנֹּחַ וְעַד אַבְרָהָם אָבִינוּ‏ ,
לְהוֹדִיעַ כַּמָּה אֶרֶךְ אַפַּיִם לְפָנָיו,
שֶׁכָּל הַדּוֹרוֹת הָיוּ מַכְעִיסִים לְפָנָיו,
עַד שֶׁבָּא אַבְרָהָם אָבִינוּ וְקִבֵּל שְׂכַר כֻּלָּם.

5-3 משנה ג
עֲשָׂרָה נִסְיוֹנוֹת נִתְנַסָּה אַבְרָהָם אָבִינוּ,
וְעָמַד בְּכֻלָּם,
לְהוֹדִיעַ כַּמָּה הִיא‏ חִבָּתוֹ שֶׁלְּאַבְרָהָם אָבִינוּ.

5-4 משנה ד
עֲשָׂרָה נִסִּים נֶעֱשׂוּ לַאֲבוֹתֵינוּ בְמִצְרַיִם,
וַעֲשָׂרָה עַל הַיָּם.
עֶשֶׂר מַכּוֹת הֵבִיא הַקָּדוֹשׁ בָּרוּךְ הוּא
עַל הַמִּצְרִיִּים‏ בְּמִצְרַיִם, וְעֶשֶׂר‏ עַל הַיָּם.
עֲשָׂרָה‏ נִסְיוֹנוֹת נִסּוּ אֲבוֹתֵינוּ‏ אֶת הַמָּקוֹם בַּמִּדְבָּר,
שֶׁנֶּאֱמַר: (במדבר יד, כב)
  וַיְנַסּוּ אֹתִי זֶה עֶשֶׂר פְּעָמִים וְלֹא שָׁמְעוּ בְקוֹלִי  .

5-5 משנה ה
עֲשָׂרָה נִסִּים נֶעֱשׂוּ בְּבֵית‏ הַמִּקְדָּשׁ:
לֹא הִפִּילָה אִשָּׁה מֵרֵיחַ בְּשַׂר קֹדֶשׁ‏ מֵעוֹלָם‏ .
וְלֹא‏ הִסְרִיחַ‏ בְּשַׂר קֹדֶשׁ‏ מֵעוֹלָם 
וְלֹא אֵרַע קֶרִי לְכֹהֵן‏ גָּדוֹל בְּיוֹם הַכִּפּוּרִים‏ .
וְלֹא נִרְאָה זְבוּב בְּבֵית הַמִּטְבְּחַיִם‏ .
וְלֹא נִמְצָא פְסוּל בָּעֹמֶר וּבִשְׁתֵּי הַלֶּחֶם וּבְלֶחֶם הַפָּנִים‏ .
וְלֹא כִבּוּ גְשָׁמִים אֶת‏ הַמַּעֲרָכָה.
וְלֹא נָצְחָה‏ הָרוּחַ אֶת עַמּוּד עָשָׁן‏ .
עוֹמְדִים צְפוּפִין וּמִשְׁתַּחֲוִים רְווּחִים‏ .
וְלֹא הִזִּיק נָחָשׁ וְעַקְרָב בִּירוּשָׁלַיִם‏ .
וְלֹא אָמַר אָדָם לַחֲבֵרוֹ:
צַר לִי הַמָּקוֹם שֶׁאָלִין בִּירוּשָׁלַיִם.

5-6 משנה ו

עֲשָׂרָה דְבָרִים‏ נִבְרְאוּ בֵין הַשְּׁמָשׁוֹת:
פִּי הָאָרֶץ,
וּפִי הַבְּאֵר,
וּפִי הָאָתוֹן,
וְהַקֶּשֶׁת,
וְהַמָּן,
וְהַמַּטֶּה,
וְהַשָּׁמִיר,
וְהַכְּתָב,
וְהַמִּכְתָּב,
וְהַלּוּחוֹת‏ .
וְיֵשׁ אוֹמְרִים:
אַף הַמַּזִּיקִין‏ ,
וּקְבוּרָתוֹ שֶׁלְּמֹשֶׁה,
וְאֵילוֹ‏ שֶׁלְּאַבְרָהָם אָבִינוּ‏ .
וְיֵשׁ אוֹמְרִים:
אַף צָבָת בִּצְבַת הָ עֲשׂוּיָה‏ 

5-7 משנה ז
שִׁבְעָה דְבָרִים בַּגֹּלֶם וְשִׁבְעָה בֶחָכָם.
הֶחָכָם אֵינוּ מְדַבֵּר לִפְנֵי‏ מִי שֶׁהוּא גָּדוֹל מִמֶּנּוּ‏ ,
וְאֵינוּ נִכְנָס לְתוֹךְ דִּבְרֵי חֲבֵרוֹ,
וְאֵינוּ נִבְהָל לְהָשִׁיב.
שׁוֹאֵל כַּהֲלָכָה וּמֵשִׁיב כָּעִנְיָן‏ ,
וְאוֹמֵר עַל רִאשׁוֹן רִאשׁוֹן וְעַל אַחֲרוֹן אַחֲרוֹן.
וְעַל מַה שֶּׁלֹּא שָׁמַע, אוֹמֵר   לֹא שָׁמַעְתִּי  ,
וּמוֹדֶה עַל הָאֱמֶת.
וְחִלּוּפֵיהֶן בַּגֹּלֶם.

5-8 משנה ח
שִׁבְעָה מִינֵי פֻרְעָנִיּוֹת בָּאִים
עַל שִׁבְעָה גוּפֵי עֲבֵרוֹת‏ :
מִקְצָתָן מְעַשְּׂרִין וּמִקְצָתָן אֵינָן‏ סופר: שֶׁאֵינָן‏ מְעַשְּׂרִין,
רָעָב שֶׁלַּבַּצֹּרֶת בָּא,
מִקְצָתָן רְעֵבִים וּמִקְצָתָן שְׂבֵעִים.
גָּמְרוּ שֶׁלֹּא לְעַשֵּׂר,
רָעָב שֶׁלִּמְהוּמָה וְשֶׁלַּבַּצֹּרֶת סופר: וְשֶׁל בַּצוֹרָה‏ פרמה: וְשֶׁלְמָצוֹר בָּא.
וְשֶׁלֹּא לִטּוֹל חַלָּה‏ ,
רָעָב שֶׁלְּכַלָּיָה בָא.
דֶּבֶר בָּא לָעוֹלָם
עַל הַמִּיתוֹת הָאֲמוּרוֹת בַּתּוֹרָה
שֶׁלֹּא נִמְסְרוּ לְבֵית דִּין,
וְעַל פֵּרוֹת שְׁבִיעִית.
חֶרֶב בָּאָה סופר: בָּא‏ לָעוֹלָם עַל עִנּוּי הַדִּין,
וְעַל עִוּוּת הַדִּין,
וְעַל הַמּוֹרִים בַּתּוֹרָה שֶׁלֹּא כַהֲלָכָה.


5-9 משנה ט
חַיָּה רָעָה בָאָה‏ לָעוֹלָם עַל שְׁבוּעַת שָׁוְא,
וְעַל חִלּוּל הַשֵּׁם.
גָּלוּת בָּאָה סופר: בָּא‏ לָעוֹלָם עַל‏ עֲבוֹדָה זָרָה,
וְעַל גִּלּוּי עֲרָיוֹת, וְעַל שְׁפִיכוּת דָּמִים,
וְעַל הַשְׁמֵט‏ הָאָרֶץ.
 בְּאַרְבָּעָה פְרָקִים הַדֶּבֶר מְרֻבֶּה:
בָּרְבִיעִית וּבַשְּׁבִיעִית, וּבְמוֹצָאֵי שְׁבִיעִית,
וּבְמוֹצָאֵי הֶחָג שֶׁבְּכָל שָׁנָה.
בָּרְבִיעִית, מִפְּנֵי מַעְשֵׂר עָנִי שֶׁבַּשְּׁלִישִׁית 
בַּשְּׁבִיעִית‏ , מִפְּנֵי מַעְשֵׂר עָנִי שֶׁבַּשִּׁשִּׁית 
בְּמוֹצָאֵי שְׁבִיעִית, מִפְּנֵי פֵרוֹת שְׁבִיעִית 
<וּ‏ >בְמוֹצָאֵי הֶחָג שֶׁבְּכָל שָׁנָה,
מִפְּנֵי גֶזֶל מַתְּנוֹת‏ עֲנִיִּים.

5-10 משנה י
אַרְבַּע מִדּוֹת בָּאָדָם:
הָאוֹמֵר   שֶׁלִּי שֶׁלִּי וְשֶׁלָּךְ שֶׁלָּךְ  ,
זוֹ‏ מִדָּה בֵינוֹנִית.
וְיֵשׁ אוֹמְרִים:
 זוֹ‏ מִדַּת סְדוֹם.
  שֶׁלִּי שֶׁלָּךְ וְשֶׁלָּךְ שֶׁלִּי  , עַם הָאָרֶץ 
  שֶׁלִּי וְשֶׁלָּךְ שֶׁלָּךְ‏   , חָסִיד,
  שֶׁלָּךְ וְשֶׁלִּי שֶׁלִּי‏   , רָשָׁע.

5-11 משנה יא
אַרְבַּע <מִדּוֹת‏ > בַּדֵּעוֹת‏ :
נוֹחַ לִכְעֹס וְנוֹחַ לִרְצוֹת,
יָצָא שְׂכָרוֹ בַהֶפְסֵדוֹ ‏

5-12 משנה יב
אַרְבַּע מִדּוֹת בַּלְּמֵידִים‏ :
מְמַהֵר לִשְׁמוֹעַ וּמְמַהֵר לְאַבֵּד,
יָצָא שְׂכָרוֹ בַהֶפְסֵדוֹ 
קָשֶׁה לִשְׁמֹעַ וְקָשֶׁה לְאַבֵּד,
יָצָא הֶפְסֵדוֹ בִשְׂכָרוֹ 
מְמַהֵר לִשְׁמֹעַ וְקָשֶׁה לְאַבֵּד,
חָכָם קָשֶׁה לִשְׁמֹעַ וּמְמַהֵר לְאַבֵּד,
זֶה חֵלֶק רַע.

5-13 משנה יג
אַרְבַּע מִדּוֹת בְּנוֹתְנֵי צְדָקָה:
רוֹצֶה שֶׁיִּתֵּן וְאַל יִתְּנוּ אֲחֵרִים,
עֵינוֹ רָעָה בְשֶׁלַּאֲחֵרִים 
 שֶׁיִּתְּנוּ‏ אֲחֵרִים וְהוּא לֹא יִתֵּן,
עֵינוֹ רָעָה בְשֶׁלּוֹ 
 שֶׁ יִּתֵּן‏ וְיִתְּנוּ אֲחֵרִים,
חָסִיד 
אַל‏ יִתֵּן וְאַל יִתְּנוּ אֲחֵרִים,
רָשָׁע.

5-14 משנה יד
אַרְבַּע מִדּוֹת בְּהוֹלְכֵי לְבֵית הַמִּדְרָשׁ:
הוֹלֵךְ וְאֵינוּ עוֹשֶׂה, שְׂכַר הֲלִיכָה בְיָדוֹ 
עוֹשֶׂה וְאֵינוּ הוֹלֵךְ, שְׂכַר מַעֲשֶׂה בְיָדוֹ 
הוֹלֵךְ וְעוֹשֶׂה, חָסִיד 
לֹא הוֹלֵךְ וְלֹא עוֹשֶׂה, רָשָׁע.

5-15 משנה טו
אַרְבַּע מִדּוֹת בְּיוֹשְׁבֵי‏ לִפְנֵי חֲכָמִים:
סְפוֹג, מַשְׁפֵּךְ, וּמְשַׁמֶּרֶת, וְנָפָה.
סְפוֹג, שֶׁהוּא סוֹפֵג אֶת הַכֹּל 
מַשְׁפֵּךְ, שֶׁהוּא מַכְנִיס בָּזוֹ וּמוֹצִיא בָזוֹ 
מְשַׁמֶּרֶת, שֶׁהִיא מוֹצִיאָה אֶת הַיַּיִן וְקוֹלֶטֶת אֶת הַשְּׁמָרִים 
נָפָה, שֶׁהִיא מוֹצִיאָה‏ אֶת הַקֶּמַח וְקוֹלֶטֶת אֶת הַסֹּלֶת.

5-16 משנה טז
כָּל אַהֲבָה שֶׁהִיא תְּלוּיָה בְדָבָר,
בָּטֵל דָּבָר, וּבָטֵלָה‏ אַהֲבָה‏  
וְשֶׁאֵינָהּ תְּלוּיָה בְדָבָר,
אֵינָה בְטֵלָה לְעוֹלָם.
אֵי זוֹ‏ הִיא אַהֲבָה שֶׁהִיא תְלוּיָה בְדָבָר?
זוֹ‏ אַהֲבַת אַמְנוֹן וְתָמָר‏ .
וְשֶׁאֵינָה תְלוּיָה בְדָבָר?
זוֹ‏ אַהֲבַת דָּוִד וִיהוֹנָתָן‏ .

5-17 משנה יז 
כָּל מַחֲלֹקֶת‏ שֶׁהִיא לְשֵׁם שָׁמַיִם,
סוֹפָהּ לְהִתְקַיֵּם 
וְשֶׁאֵינָה לְשֵׁם שָׁמַיִם,
אֵין סוֹפָהּ לְהִתְקַיֵּם.
אֵיזוֹ הִיא מַחֲלֹקֶת שֶׁהִיא <לְשֵׁם‏ > שָׁמַיִם?
מַחֲלֹקֶת שַׁמַּי וְהֶלֵּל‏  
וְשֶׁאֵינָה לְשֵׁם שָׁמַיִם?
זוֹ מַחֲלוֹקְתּוֹ‏ שֶׁלְּקֹרַח‏ .

5-18 משנה יח
כָּל הַמְזַכֶּה אֶת הָרַבִּים,
אֵין חֵטְ‏ בָּאָה נ.א. בָּא‏ עַל יָדוֹ 
וְכָל הַמַּחֲטִיא‏ אֶת הָרַבִּים,
אֵין מַסְפִּיקִין בְּיָדוֹ לַעֲשׁוֹת תְּשׁוּבָה.
משֶׁה זָכָה וְזִכָּה אֶת הָרַבִּים,
וּזְכוּת הָרַבִּים תְּלוּיָה בוֹ, שֶׁנֶּאֱמַר: (דברים לג, כא)
צִדְקַת יי עָשָׂה, וּמִשְׁפָּטָיו עִם יִשְׂרָאֵל  ‏ .
יָרָבְעָם חָטָא וְהֶחֱטִיא אֶת הָרַבִּים,
וְחֵטְא הָרַבִּים תְּלוּיָה פרמה: תָּלוּי בּוֹ, שֶׁנֶּאֱמַר: (מלכים א טו, ל)
 עַל חַטֹּאות יָרָבְעָם‏ אֲשֶׁר חָטָא
וַאֲשֶׁר הֶחֱטִיא אֶת יִשְׂרָאֵל  .

5-19 משנה יט
כָּל‏ שֶׁיֵּשׁ בּוֹ שְׁלֹשָׁה דְבָרִים‏ ,
תַּלְמִידוֹ שֶׁלְּאַבְרָהָם אָבִינוּ‏ ,
וּשְׁלֹשָׁה דְבָרִים‏ ,
תַּלְמִידוֹ שֶׁלְּבִלְעָם.
עַיִן טוֹבָה, וְנֶפֶשׁ שְׁפָלָה, וְרוּחַ נְמוּכָה‏ ,
תַּלְמִידוֹ שֶׁלְּאַבְרָהָם 
עַיִן רָעָה, וְנֶפֶשׁ רְחָבָה‏ , וְרוּחַ גְּבוֹהָה‏ ,
תַּלְמִידוֹ שֶׁלְּבִלְעָם.
מַה‏ בֵּין תַּלְמִידוֹ סופר: תַּלְמִידָיו‏ שֶׁלְּ אַבְרָהָם אָבִינוּ‏ 
לְתַלְמִידָיו שֶׁלְּבִלְעָם?
תַּלְמִידָיו שֶׁלְּבִלְעָם יוֹרְדִים לְגֵיהִנָּם‏ ,
שֶׁנֶּאֱמַר: (תהילים נה, כד)
  וְאַתָּה אֱלהִים תּוֹרִדֵם לִבְאֵר שַׁחַת‏ ,
אַנְשֵׁי דָמִים וּמִרְמָה לֹא יֶחֱצוּ יְמֵיהֶם,
וַאֲנִי אֶבְטַח בָּךְ  .
אֲבָל תַּלְמִידָיו שֶׁלְּאַבְרָהָם אָבִינוּ
יוֹרְשִׁים לְגַן עֵדֶן‏ ,
שֶׁנֶּאֱמַר: (משלי ח, כא)
  לְהַנְחִיל אֹהֲבַי‏ יֵשׁ, וְאֹצְרֹתֵיהֶם אֲמַלֵּא.  

5-20 משנה כ
יְהוּדָה בֶן תֵּימָה‏ אוֹמֵר:
הֱוֵי עַז כַּנָּמֵר,
וְקַל כַּנֶּשֶׁר,
וְרָץ כַּצְּבִי,
וְגִבּוֹר כָּאֲרִי,
לַעֲשׁוֹת רְצוֹן אָבִיךָ שֶׁבַּשָּׁמַיִם.
הוּא הָיָה אוֹמֵר:
עַז פָּנִים לְגֵיהִנָּם,
וּבוֹשׁ‏ פָּנִים לְגַן עֵדֶן.
יְהִי רָצוֹן מִלְּפָנֶיךָ יי אֱלֹהֵינוּ וֵאלֹהֵי אֲבוֹתֵינוּ,
שֶׁתִּבְנֶה עִירָךְ בְּיָמֵינוּ,
וְתִתֵּן‏ חֶלְקֵנוּ בְּתוֹרָתֶךָ.

5-21 משנה כא‏ 
הוּא הָיָה אוֹמֵר: בֶּן חָמֵשׁ שָׁנִים לַמִּקְרָא,
בֶּן עֶשֶׂר לַמִּשְׁנָה, בֶּן שְׁלשׁ עֶשְׂרֵה לַמִּצְוֹת,
 בֶּן חֲמֵשׁ עֶשְׂרֵה לַתַּלְמוּד, בֶּן שְׁמוֹנֶה עֶשְׂרֵה לַחֻפָּה,
 בּן עֶשְׂרִים לִרְדּוֹף, בֶּן שְׁלשִׁים לַכֹּחַ,
בֶּן אַרְבָּעִים לַבִּינָה, בֶּן חֲמִשִּׁים לָעֵצָה, בֶּן שִׁשִּׁים לַזִקְנָה, 
בֶּן שִׁבְעִים לַשֵּׂיבָה, בֶּן שְׁמוֹנִים לַגְּבוּרָה,
בֶּן תִּשְׁעִים לָשׁוּחַ, בֶּן מֵאָה כְּאִלּוּ מֵת וְעָבַר וּבָטֵל מִן הָעוֹלָם

5-22 משנה כב
בֶּן בַּגבַּג‏ אוֹמֵר:
הֲפֹךְ בָּהּ וְהַפֵּךְ בָּהּ‏,
דְּכֹלַּהּ בָּךְ וְכֹלָּךְ בַּהּ,
וּמֶנַּהּ‏ לָא תְזוּעַ,
שֶׁאֵין לָךְ‏ מִדָּה טוֹבָה מִמֶּנָּה‏ .

5-23 משנה כג
בֶּן הֵהֵא‏ אוֹמֵר:
לְפוּם‏ צַעְרָה אַגְרָה‏.
